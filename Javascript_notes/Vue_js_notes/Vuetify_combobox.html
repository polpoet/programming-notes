В компонентах есть встроенная функциональность.
combobox группирует все элементы по одному значению. В этом примере по номеру телефона а телефоны передаются одинаковые phoneClient.items. Нужно задать какой то атрибут который будет уникальным у каждого значения в массиве. И он не должен быть равен null item-value="contactid" 

<VCombobox
      v-model="phoneClient.text"
      :search-input.sync="form.phone"
      :items="phoneClient.items"
      label="Телефон"
      dense
      item-text="phone"
      item-value="contactid"
      append-icon="fas fa-search"
      :loading="phoneClient.loading"
      :disabled="!activeChatId"
      class="v-select--no-transform"
      v-debounce="text => search('phone', text)"
      @change="select"
    >
