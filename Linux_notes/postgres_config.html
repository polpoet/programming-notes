Установка постгрес

service --status-all //Посмотреть список сервисов

sudo service --status-all | grep postgres //подсветить

sudo -i -u postgres //Переключение на учетную запись postgres

psql // доступ к командной строке Postgres

postgres=# \q // Закройте командную строку PostgreSQL с помощью следующей команды

sudo -u postgres psql // ез переключения пользователя, доступ к командной строке Postgres

//

ALTER USER postgres PASSWORD '1234'; //Добавить пароль в командной строке psql
